## Usage
### example.js
var operation = require('sailendra_calculator');

console.log(operation(4,2,'+'));  
console.log(operation(4,2,'/'));  
console.log(operation(4,2,'-'));  
console.log(operation(4,2,'%'));  
console.log(operation(4,2,'*'));  

### Output
6  
2  
2  
0  
8
